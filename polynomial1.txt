#include <stdio.h>


void createpoly(int poly[], int degree){
    int i;
    for(i=degree;i>=0;i--){
        printf("Enter the coefficient of x^%d: ", i);
        scanf("%d",&poly[i]);
    }
}

void display(int poly[], int degree){
    int i;
    for(i=degree; i>=0; i--){
        printf("%dx^%d + ",poly[i], i);
    }
}

int main(){
// #define MAX_TERMS 30;
// int poly1[MAX_TERMS], poly2[MAX_TERMS],polyresult[MAX_TERMS], i, j, k, degree1, degree2, maxdegree;
int poly1[30], poly2[30],polyresult[50], i, j, k, degree1, degree2, maxdegree;
printf("Enter the degree of first polynomial: ");
scanf("%d", &degree1);
createpoly(poly1, degree1);

printf("Enter the degree of second polynomial: ");
scanf("%d", &degree2);
createpoly(poly2, degree2);

printf("printing first polynomial: ");
display(poly1, degree1);
printf("\n ");
printf("printing second polynomial: ");
display(poly2, degree2);
printf("\n ");


if(degree1>=degree2){
    maxdegree=degree1;
}else{maxdegree=degree2;}

for(i=maxdegree; i>=0; i--){
    polyresult[i]=poly1[i]+poly2[i];
}
printf("priting answer");
display(polyresult, maxdegree);
}
