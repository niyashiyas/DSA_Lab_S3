#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>
int n=5;
int cqueue[5] = {0};
int front = -1;
int rear = -1;

bool empty(){
    if(front == (-1)){
        return true;
    }
    else{
        return false;
    }
} 

void enqueue(int item){
    if(empty(true)){
            front=rear=0;
            cqueue[front] = item;
    }   
    else{
        if((rear+1)%n != front){
            rear = (rear+1) % n;
            cqueue[rear] = item;
        }
        else{
            printf("No space");
        }   
    }

}


void dequeue(){
    if(empty(true))
    {
        printf("nothing to delete");
    }
    else{
        if(front!=rear){
            int temp = cqueue[front];
            printf("%d item is deleted", temp);
            front=(front+1)%n;
        }
        else{
            int temp = cqueue[front];
            printf("%d item is deleted", temp);
            front=rear=-1;
        }
    }
}

void display(){
    int itr;
    if(empty(true)){
     printf("empty");  
     
    }
    else{
        for(itr=front; itr!=rear; itr=(itr+1)%n){
        printf("%d ", cqueue[itr]);
        }
        printf("%d ", cqueue[itr]);
    }

}

void main() {
  int choice, data;
  
  while(1){
  printf("\nEnter the operation: \n1)Enqueue 2)Dequeue 3)Display 4)Exit\n");
  scanf("%d", &choice);
  switch (choice) {
      case 1:
      printf("Enter the number to be inserted: ");
      scanf("%d", &data);
      enqueue(data);
      break;
      
      case 2:
      dequeue();
      break;
      
      case 3 :
      display();
      break;
      
      case 4:
      exit(0);
      
      default:
      printf("type a valid response");
      
   }
  }
}
